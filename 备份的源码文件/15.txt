//用范围裁剪数组
using System.Text;

int[] arr = new int[10];
for (int i = 0; i < arr.Length; i++)
{
	arr[i] = i;
}

arr = arr[2..^2];
for (int i = 0; i < arr.Length; i++)
{
	Console.WriteLine(arr[i]);
}

//用范围裁剪字符串
string s = "hello";
for (int i = 0; i < s.Length; i++)
{
	Console.WriteLine(s[i]);
}

s = s[..^2];
Console.WriteLine(s);

//StringBuild
StringBuilder builder = new StringBuilder();
for (int i = 0; i < 10; i++)
{
	builder.Append(i);
}
Console.WriteLine(builder.ToString());
